class Info : public DbEntity
{
public:
	Info() {  bindMember(); }
	Info(const Info& info)
	{
		bindMember();
		*this = info;
	}
	Info(Info&& info)
	{
		bindMember();
		*this = info;
	}

	void operator = (const Info& info)
	{
		m_id = info.m_id;
		m_time = info.m_time;
		m_age = info.m_age;
		m_grade = info.m_grade;
		m_math = info.m_math;
		m_remarks = info.m_remarks;
	}
	void operator =(Info&& info)
	{
		m_id = info.m_id;
		m_time = info.m_time;
		m_age = info.m_age;
		m_grade = info.m_grade;
		m_math = info.m_math;
		m_remarks = info.m_remarks;
	}

public:
	string m_id;
	string m_time;
	int m_age;
	double m_grade;
	double m_math;
	string m_remarks;

private:
	void bindMember()
	{
		bindTableName("info");

		bindFeild("id", &m_id);
		bindFeild("time", &m_time);
		bindFeild("age", &m_age);
		bindFeild("grade", &m_grade);
		bindFeild("math", &m_math);
		bindFeild("remarks", &m_remarks);
	}
};
